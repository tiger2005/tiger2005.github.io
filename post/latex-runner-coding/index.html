	return a!=' '&&a!='\t'&&a!='\r'&&a!='\n'&&a!='\\';

	while(OriginalText[a]==' '
		||OriginalText[a]=='\t'
		||OriginalText[a]=='\r'
		||OriginalText[a]=='\n')
		++a;
	return a;
}
